/**
* Definition for singly-linked list.
* public class ListNode {
* int val;
* ListNode next;
* ListNode(int x) { val = x; }
* }
*/
class Solution {
public ListNode reverseKGroup(ListNode head, int k) {
if(k==0||k==1)return head;
ListNode dummyhead=new ListNode(0);
dummyhead.next=head;
ListNode tail=dummyhead;
while(true){
int i=0;
ListNode node_head=head;
while(i<k&&head!=null){
head=head.next;
i++;
}
if(i==k){
ListNode cur=node_head;
ListNode pre=null;
ListNode next;
while(i>0){
next=cur.next;
cur.next=pre;
pre=cur;
cur=next;
i--;
}
tail.next=pre;
tail=node_head;
head=cur;
}else{
tail.next=node_head;
return dummyhead.next;
}
}
}
}
